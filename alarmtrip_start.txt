#!/bin/sh
#/etc/init.d/alarmtrip
export PATH=$PATH:/home/pi/alarmtrip
export NODE_PATH=$NODE_PATH:/usr/local/lib/node_modules

case "$1" in
start)
exec forever --sourceDir=/home/pi/alarmtrip -p /home/pi/alarmtrip alarmtrip.js  #scriptarguments
;;
stop)
exec forever stop --sourceDir=/home/pi/alarmtrip alarmtrip.js
;;
*)
echo "Usage: /etc/init.d/alarmtrip {start|stop}"
exit 1
;;
esac
exit 0
